Mátrix – beolvasás és szimmetria

int[,] fájlból olvasás · szimmetria-ellenőrzés — hiányzik a portfólióból
01
Mátrix beolvasása fájlból
Szöveges fájlból (soronként, szóközzel elválasztott számok) n×n-es kétdimenziós tömbbe olvassa be az adatokat.
static int[,] Beolvas(string fajl)
{
    string[] sorok = File.ReadAllLines(fajl);
    int n = sorok.Length;
    int[,] M = new int[n, n];

    for (int i = 0; i < n; i++)
    {
        string[] adatok = sorok[i].Split(' ');
        for (int j = 0; j < n; j++)
            M[i, j] = int.Parse(adatok[j]);
    }
    return M;
}

/* Példa bemeneti fájl (matrix.txt):
   1 2 3
   2 5 6
   3 6 9

02
Szimmetria-ellenőrzés
Egy mátrix szimmetrikus, ha M[i,j] == M[j,i] minden i,j párra. Csak a főátló feletti elemeket kell vizsgálni.
static bool Szimmetrikus(int[,] M)
{
    int n = M.GetLength(0);
    for (int i = 0; i < n; i++)
        for (int j = i + 1; j < n; j++)
            if (M[i, j] != M[j, i])
                return false;
    return true;
}

Teljes konzolprogram
A két metódus együtt: beolvasás + szimmetria kiírása a konzolra.
using System;
using System.IO;

class Program
{
    static int[,] Beolvas(string fajl)
    {
        string[] sorok = File.ReadAllLines(fajl);
        int n = sorok.Length;
        int[,] M = new int[n, n];
        for (int i = 0; i < n; i++)
        {
            string[] adatok = sorok[i].Split(' ');
            for (int j = 0; j < n; j++)
                M[i, j] = int.Parse(adatok[j]);
        }
        return M;
    }

    static bool Szimmetrikus(int[,] M)
    {
        int n = M.GetLength(0);
        for (int i = 0; i < n; i++)
            for (int j = i + 1; j < n; j++)
                if (M[i, j] != M[j, i])
                    return false;
        return true;
    }

    static void Main()
    {
        int[,] M = Beolvas("matrix.txt");
        Console.WriteLine(Szimmetrikus(M)
            ? "A mátrix szimmetrikus."
            : "A mátrix NEM szimmetrikus.");
    }
}
