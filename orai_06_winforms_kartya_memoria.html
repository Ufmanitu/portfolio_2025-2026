Button[,] · Timer · páros egyezés — hiányzik a portfólióból (a #18 konzol változat!)
ℹ️ A portfólió #18-as feladata konzol alapú kártyajáték. Ez a fájl a WinForms Button-grid változatot mutatja, ahol egy Timer gondoskodik a nem egyező kártyák visszafordításáról.

Form1.cs – teljes WinForms változat
4×4-es Button rács, véletlenszerű számkiosztás, Timer-alapú visszafordítás, páros egyezés kezelése.
public partial class Form1 : Form
{
    Button[,] gombok;
    int[,]    ertekek;
    int sor = 4, oszlop = 4;

    Button elso    = null;
    Button masodik = null;
    Timer  t = new Timer();

    public Form1()
    {
        InitializeComponent();
        t.Interval = 800;
        t.Tick    += T_Tick;
        UjJatek();
    }

    private void UjJatek()
    {
        panel1.Controls.Clear();
        gombok  = new Button[sor, oszlop];
        ertekek = new int[sor, oszlop];

        List<int> lista = new List<int>();
        for (int i = 1; i <= (sor * oszlop) / 2; i++)
        {
            lista.Add(i);
            lista.Add(i);
        }

        Random r = new Random();
        lista = lista.OrderBy(x => r.Next()).ToList();

        int index = 0;

        for (int i = 0; i < sor; i++)
        {
            for (int j = 0; j < oszlop; j++)
            {
                ertekek[i, j]        = lista[index++];
                gombok[i, j]         = new Button();
                gombok[i, j].Width   = 60;
                gombok[i, j].Height  = 60;
                gombok[i, j].Left    = j * 62;
                gombok[i, j].Top     = i * 62;
                gombok[i, j].Font    = new Font("Arial", 18, FontStyle.Bold);
                gombok[i, j].Click  += Gomb_Click;

                panel1.Controls.Add(gombok[i, j]);
            }
        }
    }

    private void Gomb_Click(object sender, EventArgs e)
    {
        Button b = sender as Button;

        int x = b.Top  / 62;
        int y = b.Left / 62;

        b.Text    = ertekek[x, y].ToString();
        b.Enabled = false;

        if (elso == null)
        {
            elso = b;
        }
        else
        {
            masodik = b;

            if (elso.Text == masodik.Text)
            {
                elso = null;
                masodik = null;
            }
            else
            {
                t.Start();
            }
        }
    }

    private void T_Tick(object sender, EventArgs e)
    {
        t.Stop();
        elso.Text    = "";
        masodik.Text = "";
        elso.Enabled    = true;
        masodik.Enabled = true;
        elso    = null;
        masodik = null;
    }
}
⚠️ WinForms projekt. Szükséges: panel1 az Designerben. A gombok dinamikusan kerülnek a panelre. Az UjJatek() egy menügombhoz is köthető az újrakezdéshez.
