Fordított szám
Pozitív egész szám számjegyeit fordított sorrendbe írjuk. Felesleges vezető nullákat elhagyjuk. (HF)
▼

using System;

namespace ForditottSzam
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Adja meg a pozitív egész számot: ");
            string input = Console.ReadLine().Trim();

            // Ha negatív előjel van, kezeljük
            bool negativ = input.StartsWith("-");
            if (negativ) input = input.Substring(1);

            char[] chars = input.ToCharArray();
            Array.Reverse(chars);
            string fordított = new string(chars).TrimStart('0');

            if (fordított == "") fordított = "0";
            if (negativ && fordított != "0") fordított = "-" + fordított;

            Console.WriteLine($"Fordított szám: {fordított}");
        }
    }
}

Szorzatok száma – páros szorzatú részsorozatok

          
Hány nem-üres részsorozatnak (sorrendtartó, nem feltétlenül szomszédos) páros a szorzata? (HF)

        

        ▼
      

      

        

          
C# · KONZOL


using System;

namespace SzorzatokSzama
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Elemek száma (n): ");
            int n = int.Parse(Console.ReadLine());
            int[] tomb = new int[n];

            for (int i = 0; i < n; i++)
            {
                Console.Write($"  {i + 1}. elem: ");
                tomb[i] = int.Parse(Console.ReadLine());
            }

            // Kulcsfelismerés:
            // Összes nem-üres részsorozat: 2^n - 1
            // Páratlan szorzatú ↔ MINDEN eleme páratlan → 2^k - 1 (k = páratlan elemek száma)
            // Páros szorzatú = (2^n - 1) - (2^k - 1) = 2^n - 2^k

            int paratlanDarab = 0;
            foreach (int x in tomb)
                if (x % 2 != 0) paratlanDarab++;

            long osszes   = (1L << n) - 1;
            long mindParat = (1L << paratlanDarab) - 1;
            long parosSzorzat = osszes - mindParat;

            Console.WriteLine($"\nPáros szorzatú részsorozatok száma: {parosSzorzat}");
        }
    }
}

Háromszög – teljes elemzés (oldalak + szögek)

          
Ha háromszög, akkor egyenlő oldalú / szárú / általános, és hegyesszögű / derékszögű / tompaszögű. (Dolgozat)

        

        ▼

using System;

namespace HaromszogElemzes
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("a oldal: ");
            double a = double.Parse(Console.ReadLine().Replace('.',','));
            Console.Write("b oldal: ");
            double b = double.Parse(Console.ReadLine().Replace('.',','));
            Console.Write("c oldal: ");
            double c = double.Parse(Console.ReadLine().Replace('.',','));

            if (a <= 0 || b <= 0 || c <= 0 ||
                !(a + b > c && a + c > b && b + c > a))
            {
                Console.WriteLine("Ezek nem alkothatnak háromszöget.");
                return;
            }

            Console.WriteLine("Érvényes háromszög!\n");

            if (a == b && b == c)
                Console.WriteLine("Oldalak: Szabályos (egyenlő oldalú) háromszög");
            else if (a == b || b == c || a == c)
                Console.WriteLine("Oldalak: Egyenlő szárú háromszög");
            else
                Console.WriteLine("Oldalak: Általános (különböző oldalú) háromszög");

            double cosA = (b*b + c*c - a*a) / (2*b*c);
            double cosB = (a*a + c*c - b*b) / (2*a*c);
            double cosC = (a*a + b*b - c*c) / (2*a*b);

            double angA = Math.Acos(cosA) * 180 / Math.PI;
            double angB = Math.Acos(cosB) * 180 / Math.PI;
            double angC = Math.Acos(cosC) * 180 / Math.PI;

            Console.WriteLine($"Szögek: α = {angA:F2}°,  β = {angB:F2}°,  γ = {angC:F2}°");

            const double EPS = 1e-9;
            if (Math.Abs(cosA) < EPS || Math.Abs(cosB) < EPS || Math.Abs(cosC) < EPS)
                Console.WriteLine("Szögek: Derékszögű háromszög");
            else if (cosA < 0 || cosB < 0 || cosC < 0)
                Console.WriteLine("Szögek: Tompaszögű háromszög");
            else
                Console.WriteLine("Szögek: Hegyesszögű háromszög");
        }
    }
}

10 egész szám – minimum és maximum

        ▼

using System;

namespace TizSzamMinMax
{
    class Program
    {
        static int[] Beolvas()
        {
            int[] tomb = new int[10];
            Console.WriteLine("Adjon meg 10 egész számot:");
            for (int i = 0; i < 10; i++)
            {
                Console.Write($"  {i + 1,2}. szám: ");
                tomb[i] = int.Parse(Console.ReadLine());
            }
            return tomb;
        }

        static void Kiir(int min, int max)
        {
            Console.WriteLine($"\nLegkisebb: {min,-10}  Legnagyobb: {max}");
        }

        static void Main(string[] args)
        {
            int[] tomb = Beolvas();

            int min = tomb[0], max = tomb[0];
            foreach (int x in tomb)
            {
                if (x < min) min = x;
                if (x > max) max = x;
            }

            Kiir(min, max);
        }
    }
}

        #10
        

          
Forint címletező

          
Csak pozitív, 5-tel osztható összeget fogad el. Mindig a lehető legnagyobb forint-címletet adja ki. (Dolgozat)

        

        ▼
      

      

        

          
C# · KONZOL


using System;

namespace Cimletező
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] cimletek = { 20000, 10000, 5000, 2000, 1000,
                               500, 200, 100, 50, 20, 10, 5 };

            Console.Write("Összeg (Ft): ");
            string input = Console.ReadLine();

            if (!int.TryParse(input, out int osszeg) || osszeg <= 0 || osszeg % 5 != 0)
            {
                Console.WriteLine("Érvénytelen! Pozitív, 5-tel osztható egész szám kell.");
                return;
            }

            Console.Write($"{osszeg} Ft = ");
            int maradek = osszeg;
            bool elso = true;

            foreach (int cimlet in cimletek)
            {
                int db = maradek / cimlet;
                if (db > 0)
                {
                    if (!elso) Console.Write(" + ");
                    Console.Write($"{db}×{cimlet}");
                    maradek -= db * cimlet;
                    elso = false;
                }
            }
            Console.WriteLine(" Ft");
        }
    }
}

        #11
        

          
Prímszám vagy összetett szám?

          
Pozitív egész számról eldöntjük, prím (törzsszám) vagy összetett-e. (Dolgozat)

        

        ▼
      

      

        

          
C# · KONZOL


using System;

namespace PrimSzam
{
    class Program
    {
        static bool PrimE(int n)
        {
            if (n < 2) return false;
            if (n == 2) return true;
            if (n % 2 == 0) return false;
            for (int i = 3; i * i <= n; i += 2)
                if (n % i == 0) return false;
            return true;
        }

        static void Main(string[] args)
        {
            Console.Write("Adjon meg egy pozitív egész számot: ");
            int n = int.Parse(Console.ReadLine());

            if (n < 1) { Console.WriteLine("A szám nem pozitív."); return; }

            if (PrimE(n))
                Console.WriteLine($"{n} prímszám (törzsszám).");
            else
                Console.WriteLine($"{n} összetett szám.");
        }
    }
}

Tömb-tételek GUI alkalmazás

          
N-elemű tömb generálása [-N; 3N] között. ComboBox: Eldöntés / Kiválasztás / Összegzés / Egyszerű cserés rendezés. (feladat.txt)

        

        ▼
      

      

        

          
C# · WINFORMS · Form1.cs


using System;
using System.Drawing;
using System.Windows.Forms;

namespace TombFeladatok
{
    public partial class Form1 : Form
    {
        private int    n;
        private int[]  tomb;
        private Random rnd = new Random();

        /* DESIGNER VEZÉRLŐK:
           NumericUpDown nudN  (1–20, érték: 5)
           Button btnGenerál
           ComboBox cboFeladat  (Items: l.lent)
           NumericUpDown nudInput  (határok futáskor frissülnek)
           Button btnVégrehajt
           ListBox lstRendezetlen, lstRendezett
           Label lblEredmény                        */

        public Form1()
        {
            InitializeComponent();
            cboFeladat.Items.AddRange(new object[]
            {
                "Eldöntés tétele", "Kiválasztás tétele",
                "Összegzés tétele", "Egyszerű cserés rendezés"
            });
            cboFeladat.SelectedIndex = 0;
        }

        private void btnGenerál_Click(object sender, EventArgs e)
        {
            n    = (int)nudN.Value;
            tomb = new int[n];
            for (int i = 0; i < n; i++)
                tomb[i] = rnd.Next(-n, 3 * n + 1);
            nudInput.Minimum = -n; nudInput.Maximum = 3 * n;
            lblEredmény.Text = $"Tömb generálva ({n} elem)";
        }

        private void btnVégrehajt_Click(object sender, EventArgs e)
        {
            switch (cboFeladat.SelectedIndex)
            {
                case 0: Eldöntes();  break;
                case 1: Kivalasztas(); break;
                case 2: Osszegzes(); break;
                case 3: CseresRendezes(); break;
            }
        }

        private void Eldöntes()
        {
            int ker = (int)nudInput.Value; bool t = false;
            for (int i = 0; i < n && !t; i++) if (tomb[i] == ker) t = true;
            MessageBox.Show(t ? $"{ker} megtalálható." : $"{ker} nincs a tömbben.");
        }

        private void Kivalasztas()
        {
            int ker = (int)nudInput.Value; bool t = false; int idx = -1;
            for (int i = 0; i < n && !t; i++) if (tomb[i] == ker) { t = true; idx = i; }
            lblEredmény.Text = t ? $"{ker} indexe: {idx}" : $"{ker} nincs a tömbben.";
        }

        private void Osszegzes()
        {
            long s = 0; foreach (int x in tomb) s += x;
            lblEredmény.Text = $"Összeg: {s}";
        }

        private void CseresRendezes()
        {
            lstRendezetlen.Items.Clear();
            foreach (int x in tomb) lstRendezetlen.Items.Add(x);
            int[] r = (int[])tomb.Clone();
            for (int i = 0; i < r.Length - 1; i++)
                for (int j = i + 1; j < r.Length; j++)
                    if (r[i] > r[j]) { int tmp = r[i]; r[i] = r[j]; r[j] = tmp; }
            lstRendezett.Items.Clear();
            foreach (int x in r) lstRendezett.Items.Add(x);
            lblEredmény.Text = "Rendezés kész.";
        }
    }
}

using System;

namespace Konyvek

    class Konyv

        public string Cim     { get; set; }
        public int    Ev      { get; set; }
        public string ISBN    { get; set; }

        public Konyv(int s, string sz, string c, int e, string i)
        { Sorszam=s; Szerzo=sz; Cim=c; Ev=e; ISBN=i; }
    }

    public partial class Form1 : Form
    {
        private List<Konyv> Konyvek = new List<Konyv>();

        /* DESIGNER: DataGridView dgv (Dock=Fill, AutoSizeColumnsMode=Fill, ReadOnly=true)  */

        public Form1() { InitializeComponent(); BetöltKönyvek(); FeltöltGrid(); }

        private void BetöltKönyvek()
        {
            foreach (string sor in File.ReadAllLines("konyv.txt", System.Text.Encoding.UTF8))
            {
                if (sor.StartsWith("#")) continue;

Folyamatosan változó háttérszín

          
Az RGB paletta alapján az ablak háttérszíne folyamatosan, átmenetesen változik Timer segítségével. (WinForm Designer #4)

C# · WINFORMS · Form1.cs


using System;
using System.Drawing;
using System.Windows.Forms;

namespace ValtozoBatterszin
{
    public partial class Form1 : Form
    {
        /* DESIGNER: Timer tmrSzin (Interval=30, Enabled=true) */

        private double hue = 0.0;

        public Form1() { InitializeComponent(); tmrSzin.Start(); }

        private void tmrSzin_Tick(object sender, EventArgs e)
        {
            hue = (hue + 0.5) % 360.0;
            BackColor = HsvToRgb(hue, 1.0, 1.0);
            Text = $"Háttérszín – {BackColor.R}, {BackColor.G}, {BackColor.B}";
        }

        private static Color HsvToRgb(double h, double s, double v)
        {
            int    hi = (int)(h / 60) % 6;
            double  f = h / 60 - Math.Floor(h / 60);
            double  p = v * (1 - s), q = v * (1 - f * s),
                    t = v * (1 - (1 - f) * s);
            double r, g, b;
            switch (hi)
            {
                case 0: r=v; g=t; b=p; break; case 1: r=q; g=v; b=p; break;
                case 2: r=p; g=v; b=t; break; case 3: r=p; g=q; b=v; break;
                case 4: r=t; g=p; b=v; break; default: r=v; g=p; b=q; break;
            }
            return Color.FromArgb((int)(r*255),(int)(g*255),(int)(b*255));
        }
    }
}

C# · WINFORMS · Form1.cs


using System;
using System.Collections.Generic;
using System.Drawing;
using System.Linq;
using System.Windows.Forms;

namespace KartyaMemoria
{
    public partial class Form1 : Form
    {
        /* DESIGNER:
           ComboBox cboMeret   (Items: "4×4","4×6","6×6","6×8"; SelectedIndex=0)
           Button   btnUjJatek "Új játék"
           Panel    pnlJatek   (játéktér, pl. 500×400)
           Label    lblAllapot                                  */

        private int      sorok, oszlopok;
        private Button[] kartyak;
        private int[]    ertekek;
        private int      elsoIndex    = -1;
        private int      masodikIndex = -1;
        private bool     varakozas    = false;
        private Timer    tmr          = new Timer { Interval = 900 };

        public Form1()
        {
            InitializeComponent();
            cboMeret.Items.AddRange(new object[] { "4×4", "4×6", "6×6", "6×8" });
            cboMeret.SelectedIndex = 0;
            tmr.Tick += TmrTick;
        }

        private void btnUjJatek_Click(object sender, EventArgs e) => UjJatek();

        private void UjJatek()
        {
            switch (cboMeret.SelectedIndex)
            {
                case 0: sorok=4; oszlopok=4; break;
                case 1: sorok=4; oszlopok=6; break;
                case 2: sorok=6; oszlopok=6; break;
                default: sorok=6; oszlopok=8; break;
            }
            int db = sorok * oszlopok;
            var lista = new List<int>();
            for (int i = 1; i <= db / 2; i++) { lista.Add(i); lista.Add(i); }
            var rnd = new Random();
            ertekek = lista.OrderBy(_ => rnd.Next()).ToArray();
            // ... kártyák elhelyezése pnlJatek-re (lásd teljes kód)
            elsoIndex = -1; varakozas = false;
            lblAllapot.Text = "Fordíts fel egy kártyát!";
        }

        private void Kartya_Click(object sender, EventArgs e)
        {
            if (varakozas) return;
            var btn = (Button)sender;
            int idx = (int)btn.Tag;
            if (!btn.Enabled || btn.Text != "" || idx == elsoIndex) return;
            btn.Text = ertekek[idx].ToString();
            btn.BackColor = Color.WhiteSmoke; btn.ForeColor = Color.DimGray;
            if (elsoIndex == -1) { elsoIndex = idx; lblAllapot.Text = "Fordíts fel még egyet!"; }
            else
            {
                masodikIndex = idx; varakozas = true;
                if (ertekek[elsoIndex] == ertekek[masodikIndex])
                {
                    kartyak[elsoIndex].BackColor = kartyak[masodikIndex].BackColor = Color.MediumSeaGreen;
                    kartyak[elsoIndex].Enabled   = kartyak[masodikIndex].Enabled   = false;
                    elsoIndex = -1; varakozas = false;
                    EllenorizVege();
                }
                else tmr.Start();
            }
        }

        private void TmrTick(object sender, EventArgs e)
        {
            tmr.Stop();
            kartyak[elsoIndex].Text  = kartyak[masodikIndex].Text  = "";
            kartyak[elsoIndex].BackColor = kartyak[masodikIndex].BackColor = Color.SteelBlue;
            kartyak[elsoIndex].ForeColor = kartyak[masodikIndex].ForeColor = Color.White;
            elsoIndex = -1; varakozas = false;
            lblAllapot.Text = "Fordíts fel egy kártyát!";
        }

        private void EllenorizVege()
        {
            if (kartyak.All(b => !b.Enabled))
                MessageBox.Show("Gratulálok! Minden párt megtaláltál! 🎉", "Nyertél!",
                                 MessageBoxButtons.OK, MessageBoxIcon.Information);
        }
    }
}
