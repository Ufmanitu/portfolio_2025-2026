01
Versenyző osztály és lista feltöltése
Egyszerű adatosztály tulajdonságokkal, majd generikus lista inicializálása object-initializer szintaxissal.
class Versenyzo
{
    public string Nev     { get; set; }
    public int    Pont    { get; set; }
    public bool   Ervenyes { get; set; }
}

List<Versenyzo> lista = new List<Versenyzo>()
{
    new Versenyzo() { Nev="Anna",  Pont=12, Ervenyes=true  },
    new Versenyzo() { Nev="Béla",  Pont=8,  Ervenyes=false },
    new Versenyzo() { Nev="Csaba", Pont=15, Ervenyes=true  },
    new Versenyzo() { Nev="Dóra",  Pont=6,  Ervenyes=true  }
};
02
LINQ műveletek a listán
A leggyakoribb lekérdező metódusok lambda kifejezésekkel.
// Csak az érvényes versenyzők
var ervenyesek = lista.Where(x => x.Ervenyes);

// Összes versenyző pontjainak átlaga
var atlag = lista.Average(x => x.Pont);

// Az első elem (kivételt dob, ha üres a lista)
var elso = lista.First();

// A legmagasabb pontszám
var max = lista.Max(x => x.Pont);

// Keresés névre (null-t ad vissza, ha nincs)
var keres = lista.Find(x => x.Nev == "Csaba");
